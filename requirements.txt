ultralytics>=8.4,<9.0

# PyTorch 2.7+ aarch64 wheels are Jetson-targeted CUDA builds that pull
# ~2.5 GB of NVIDIA libraries — useless on a Pi 5 and they fail to import
# without the matching libcudart on disk. Pin to <2.7 so pip picks the
# older CPU-only aarch64 wheel. (On x86_64 this constraint is also fine —
# torch 2.6 still ships full CUDA support on that platform.)
torch>=2.0,<2.7
torchaudio>=2.0,<2.7

opencv-python-headless>=4.8
sounddevice>=0.4.6
soundfile>=0.12
numpy>=1.24

# GPIO on Pi 5 needs the lgpio backend (RPi.GPIO doesn't work on Pi 5).
gpiozero>=2.0
lgpio>=0.2.2
